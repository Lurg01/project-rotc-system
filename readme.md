    2. change environment.text to .env
    3.  xamp version 3.3.0 then create database name rotc
    4. run terminal
        1. php artisan optimize
        2. create database rotc
        3. php artisan migrate:fresh --seed
        4. php artisan optimize
        5. php artisan serve
